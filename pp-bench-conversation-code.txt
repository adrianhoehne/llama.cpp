    const canvas = document.querySelector("#board");
    const ctx = canvas.getContext("2d");
    const scoreEl = document.querySelector("#score");
    const stateEl = document.querySelector("#state");
    const restartBtn = document.querySelector("#restart");

    const size = 24;
    const cells = canvas.width / size;
    const tickMs = 95;

    let snake;
    let food;
    let direction;
    let queuedDirection;
    let score;
    let gameOver;

    function reset() {
      snake = [

        { x: 7, y: 10 },
        { x: 6, y: 10 }

          x: Math.floor(Math.random() * cells),
          y: Math.floor(Math.random() * cells)

          return;

        return;

        return;

        y: head.y + direction.y

        stateEl.textContent = "Game Over";

        draw();
        return;

      ctx.fillRect(x * size + 1, y * size + 1, size - 2, size - 2);

      ctx.strokeStyle = "#17202a";
      for (let i = 0; i <= cells; i++) {
        ctx.beginPath();
        ctx.moveTo(i * size, 0);
        ctx.lineTo(i * size, canvas.height);
        ctx.stroke();

        ctx.beginPath();
        ctx.moveTo(0, i * size);
        ctx.lineTo(canvas.width, i * size);
        ctx.stroke();

      drawCell(food.x, food.y, "#ff5c7a");
      snake.forEach((part, index) => {
        drawCell(part.x, part.y, index === 0 ? "#7ee787" : "#2ea043");
      });
    }

      if (event.key === "ArrowDown") setDirection({ x: 0, y: 1 });
      if (event.key === "ArrowLeft") setDirection({ x: -1, y: 0 });
      if (event.key === "ArrowRight") setDirection({ x: 1, y: 0 });
    });

    restartBtn.addEventListener("click", reset);
    reset();

Zweitens lesen wir beim Start den Highscore aus `localStorage`, aktualisieren ihn bei Game Over und zeigen ihn neben dem aktuellen Score an.

Die wichtigsten Aenderungen sind:

```js
const highScoreEl = document.querySelector("#highscore");
let highScore = Number(localStorage.getItem("snakeHighScore") || 0);
highScoreEl.textContent = highScore;

let touchStart = null;

canvas.addEventListener("touchstart", event => {
  const touch = event.changedTouches[0];
  touchStart = { x: touch.clientX, y: touch.clientY };
}, { passive: true });

canvas.addEventListener("touchend", event => {
  if (!touchStart) return;
  const touch = event.changedTouches[0];
  const dx = touch.clientX - touchStart.x;
  const dy = touch.clientY - touchStart.y;

  if (Math.max(Math.abs(dx), Math.abs(dy)) < 24) return;

  if (Math.abs(dx) > Math.abs(dy)) {
    setDirection(dx > 0 ? { x: 1, y: 0 } : { x: -1, y: 0 });
  } else {
    setDirection(dy > 0 ? { x: 0, y: 1 } : { x: 0, y: -1 });
  }
});

function finishGame() {
  gameOver = true;
  stateEl.textContent = "Game Over";
  clearInterval(timer);

  if (score > highScore) {
    highScore = score;
    localStorage.setItem("snakeHighScore", String(highScore));
    highScoreEl.textContent = highScore;
  }
}

Assistant: Pro Tick laeuft der Code in einer festen Reihenfolge:

1. Die zuletzt vorgemerkte Richtung wird uebernommen.
2. Aus Kopfposition und Richtung wird der naechste Kopf berechnet.
3. Kollisionen werden geprueft.
4. Der neue Kopf wird vorne eingefuegt.
5. Wenn Food getroffen wurde, steigt der Score und neues Food wird platziert.
6. Wenn kein Food getroffen wurde, wird das letzte Snake-Segment entfernt.
7. Das Canvas wird neu gezeichnet.

Hindernisse passen am besten in die Kollisionserkennung. Du kannst ein Array `obstacles` einfuehren und dann pruefen:

```js
const hitsObstacle = obstacles.some(block =>
  block.x === nextHead.x && block.y === nextHead.y
);

if (hitsWall || hitsSelf || hitsObstacle) {
  finishGame();
  draw();
  return;
}
```

Level passen gut in die Food-Logik. Alle fuenf Punkte koennte das Level steigen, der Tick schneller werden und neue Hindernisse koennten entstehen. Wichtig ist, den Timer dann sauber neu zu setzen:

```js
function updateLevel() {
  const nextLevel = Math.floor(score / 5) + 1;
  if (nextLevel !== level) {
    level = nextLevel;
    clearInterval(timer);
    timer = setInterval(tick, Math.max(45, tickMs - (level - 1) * 8));
  }
}
```
